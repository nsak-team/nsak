	% --- UC-02 ----------------------------------------------------
	\textbf{UC-02} &
	\textbf{Use-Case:} Select Scenarios \newline
	\textbf{Description:} From the chosen environment, the user selects which scenarios (and related drills) shall be executed and orders them by mission priority / noise level. \newline
	\textbf{Actor:} User \newline
	\textbf{Trigger:} User wants to define a mission plan. \newline
	\textbf{Preconditions:} Environment selected (UC-0001); scenario list available. \newline
	\textbf{Main Scenario:} \newline
	1. User requests list of scenarios. \newline
	2. System displays available scenarios with metadata (impact, noise). \newline
	3. User selects one or more scenarios and sets execution order. \newline
	4. System validates selection and stores it. \newline
	\textbf{Alternative Scenarios:} No scenarios available → inform user. \newline
	\textbf{Error Scenarios:} Conflicting resources between scenarios \newline
	\textbf{Result:} Selected scenarios recorded. \newline
	\textbf{Postconditions:} Mission plan saved; ready for preparation (UC-03).\\
	\midrule
	
	% --- UC-03 ----------------------------------------------------
	\textbf{UC-03} &
	\textbf{Use-Case:} Prepare Scenario \newline
	\textbf{Description:} Load, validate and prepare the selected scenarios: collect drills, resolve dependencies, build container. \newline
	\textbf{Actor:} User \newline
	\texttt{nsak prepare <scenario>}. \newline
	\textbf{Preconditions:} Environment + scenario selection available (UC-01/02). \newline
	\textbf{Main Scenario:} \newline
	1. System fetches referenced drills and their \texttt{drill.yaml}. \newline
	2. System validates configuration (includes UC-04.3 Verify). \newline
	3. System aggregates APT/Pip dependencies and resolves conflicts. \newline
	4. System builds container image / prepares runtime. \newline
	5. System marks scenario as \emph{prepared} and logs results. \newline
	\textbf{Alternative Scenarios:} CI pipeline performs the same steps. \newline
	\textbf{Error Scenarios:} Validation or dependency failure → preparation aborted with diagnostics. \newline
	\textbf{Result:} Prepared scenario image / runtime available. \newline
	\textbf{Postconditions:} Scenario state = \emph{prepared}.\\
	\midrule

	\textbf{UC-04} &
	\textbf{Use-Case:} Use Commands (CLI) \newline
	\textbf{Description:} Unified CLI to manage scenarios (list, prepare, start, stop, verify, status, help, version). \newline
	\textbf{Actor:} User \newline
	\textbf{Trigger:} User runs \texttt{nsak <subcommand>}. \newline
	\textbf{Preconditions:} nsak installed \newline
	\textbf{Main Scenario:} \newline
	1. CLI parses args and dispatches to subsystem. \newline
	2. Command executes and writes logs. \newline
	3. CLI prints structured output and exit code. \newline
	\textbf{Alternative Scenarios:} Machine-readable output via \texttt{--json}. \newline
	\textbf{Error Scenarios:} Invalid args → help; subsystem error → non-zero exit. \newline
	\textbf{Result:} Requested action performed or error reported. \newline
	\textbf{Postconditions:} State updated accordingly.\\

	\textbf{UC-04.3} &
	\textbf{Use-Case:} Verify Scenario / Drill \newline
	\textbf{Description:} Validate \texttt{scenario.yaml}/\texttt{drill.yaml} structure and declared dependencies (syntax + semantics). \newline
	\textbf{Actor:} User \newline
	\textbf{Trigger:} \texttt{nsak verify <scenario|drill>}. \newline
	\textbf{Preconditions:} Files present and accessible. \newline
	\textbf{Main Scenario:} \newline
	1. System parses YAML and required keys. \newline
	2. System checks referenced drills and dependencies. \newline
	3. System reports validation result and diagnostics. \newline
	\textbf{Alternative Scenarios:} Verification in CI on PRs. \newline
	\textbf{Error Scenarios:} Missing files / invalid YAML → error with line/col. \newline
	\textbf{Result:} OK or detailed error. \newline
	\textbf{Postconditions:} Decision basis for preparation.\\
