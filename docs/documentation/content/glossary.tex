\newglossaryentry{RedTeam}{
  name=Red Team,
  description={The red team consists of ethical hackers employed by an organization, in an exercise they try to attack systems, networks and software by crcumvent blue team measures and activities. }
}

\newglossaryentry{BlueTeam}{
  name=Blue Team,
  description={The blue team is responsible for protective measures within an organization, in an exercise they try to detect and defend against the red team engagements. }
}

\newglossaryentry{PurpleTeam}{
  name=Purple Team,
  description={Purple teaming describes the combination or cooperation between red and blue teams, enhancing the insights in both teams. }
}

\newglossaryentry{YellowTeam}{
  name=Yellow Team,
  description={The yellow or builder team, consists of network, system and software architects and engineers. They may not have an active role during an exercise. }
}

\newglossaryentry{OrangeTeam}{
  name=Orange Team,
  description={The orange team is the cooperation of the red and yellow team, usually for sharing insights, know how and education.}
}

\newglossaryentry{GreenTeam}{
  name=Green Team,
  description={The green team is the cooperation of the blue and yellow team, for closing knowledge gaps in both teams.}
}

\newglossaryentry{WhiteTeam}{
  name=White Team,
  description={The white team consists of other possibly non-technical members of the organization, which are providing insights and requirements regarding compliance, management, analytics and operations.}
